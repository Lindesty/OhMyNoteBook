# 解构与展开

解构（Destructuring）和展开（Spread）是 ES6 引入的两个重要特性，可以简化变量赋值和对象/数组的操作。对于来自 Java/C# 的开发者，这些语法可能比较陌生，但它们能显著减少代码量。

## 对象解构

### 基本语法

```ts
const user = {
  name: "Tom",
  age: 30,
  email: "tom@example.com"
};

// 传统方式
const name = user.name;
const age = user.age;

// 解构语法
const { name, age, email } = user;
console.log(name);  // "Tom"
console.log(age);   // 30
console.log(email); // "tom@example.com"
```

### 重命名变量

```ts
const user = { name: "Tom", age: 30 };

// 将 name 重命名为 userName
const { name: userName, age: userAge } = user;
console.log(userName); // "Tom"
console.log(userAge);   // 30
```

### 默认值

当属性不存在或为 `undefined` 时，使用默认值：

```ts
const user = { name: "Tom" };

const { name, age = 25, role = "user" } = user;
console.log(name); // "Tom"
console.log(age);  // 25（使用默认值）
console.log(role); // "user"（使用默认值）
```

### 解构嵌套对象

```ts
const user = {
  name: "Tom",
  profile: {
    address: {
      city: "Beijing",
      country: "China"
    }
  }
};

const { profile: { address: { city } } } = user;
console.log(city); // "Beijing"

// 同时保留外层引用
const { profile, profile: { address } } = user;
console.log(profile);  // { address: { city: "Beijing", country: "China" } }
console.log(address);   // { city: "Beijing", country: "China" }
```

### 剩余属性

```ts
const user = {
  id: 1,
  name: "Tom",
  email: "tom@example.com",
  role: "admin"
};

const { id, ...rest } = user;
console.log(id);   // 1
console.log(rest); // { name: "Tom", email: "tom@example.com", role: "admin" }
```

### 在函数参数中使用

```ts
interface UserOptions {
  name: string;
  age?: number;
  email?: string;
  role?: string;
}

// 解构参数 + 默认值
function createUser({ name, age = 25, email = "", role = "user" }: UserOptions) {
  return { name, age, email, role };
}

createUser({ name: "Tom" });
// { name: "Tom", age: 25, email: "", role: "user" }

createUser({ name: "Jerry", age: 30, role: "admin" });
// { name: "Jerry", age: 30, email: "", role: "admin" }
```

## 数组解构

### 基本语法

```ts
const colors = ["red", "green", "blue"];

const [first, second, third] = colors;
console.log(first);  // "red"
console.log(second); // "green"
console.log(third);  // "blue"
```

### 跳过元素

```ts
const colors = ["red", "green", "blue", "yellow"];

const [first, , , fourth] = colors;
console.log(first);  // "red"
console.log(fourth); // "yellow"
```

### 默认值

```ts
const colors = ["red"];

const [first, second = "green"] = colors;
console.log(first);  // "red"
console.log(second); // "green"（使用默认值）
```

### 剩余元素

```ts
const numbers = [1, 2, 3, 4, 5];

const [first, second, ...rest] = numbers;
console.log(first);  // 1
console.log(second); // 2
console.log(rest);   // [3, 4, 5]
```

### 交换变量

```ts
let a = 1;
let b = 2;

// 传统方式
const temp = a;
a = b;
b = temp;

// 解构方式
[a, b] = [b, a];
console.log(a); // 2
console.log(b); // 1
```

### 解构函数返回值

```ts
function getCoordinates(): [number, number] {
  return [10, 20];
}

const [x, y] = getCoordinates();
console.log(x); // 10
console.log(y); // 20
```

## 展开运算符

展开运算符（Spread Operator，`...`）用于将数组或对象展开为独立元素。

### 数组展开

```ts
const arr1 = [1, 2, 3];
const arr2 = [4, 5, 6];

// 合并数组
const merged = [...arr1, ...arr2];
console.log(merged); // [1, 2, 3, 4, 5, 6]

// 在开头添加元素
const newArr = [0, ...arr1];
console.log(newArr); // [0, 1, 2, 3]

// 在中间插入
const inserted = [...arr1.slice(0, 1), 10, ...arr1.slice(1)];
console.log(inserted); // [1, 10, 2, 3]
```

### 对象展开

```ts
const defaults = {
  host: "localhost",
  port: 3000,
  timeout: 5000
};

const config = {
  ...defaults,
  port: 8080,  // 覆盖默认值
  debug: true   // 添加新属性
};

console.log(config);
// { host: "localhost", port: 8080, timeout: 5000, debug: true }
```

### 对象展开顺序

后面的属性会覆盖前面的同名属性：

```ts
const obj1 = { a: 1, b: 2 };
const obj2 = { b: 3, c: 4 };

const merged = { ...obj1, ...obj2 };
console.log(merged); // { a: 1, b: 3, c: 4 }
```

### 浅拷贝

展开运算符只进行浅拷贝：

```ts
const original = {
  name: "Tom",
  profile: { age: 30 }
};

const copy = { ...original };

copy.name = "Jerry";
console.log(original.name); // "Tom"（未受影响）

copy.profile.age = 25;
console.log(original.profile.age); // 25（被影响了！）
```

### 深拷贝方案

```ts
// 方法1：JSON（有局限性，不能处理函数、undefined、Symbol）
const deepCopy1 = JSON.parse(JSON.stringify(original));

// 方法2：structuredClone（现代浏览器支持）
const deepCopy2 = structuredClone(original);

// 方法3：递归函数
function deepClone<T>(obj: T): T {
  if (obj === null || typeof obj !== "object") {
    return obj;
  }
  if (Array.isArray(obj)) {
    return obj.map(item => deepClone(item)) as T;
  }
  const result: any = {};
  for (const key in obj) {
    if (Object.prototype.hasOwnProperty.call(obj, key)) {
      result[key] = deepClone(obj[key]);
    }
  }
  return result;
}
```

## 剩余参数

剩余参数（Rest Parameters）用于将不定数量的参数收集为数组。

### 函数参数

```ts
// 传统方式：使用 arguments
function sum() {
  return Array.from(arguments).reduce((a, b) => a + b, 0);
}

// 使用剩余参数
function sum(...numbers: number[]): number {
  return numbers.reduce((a, b) => a + b, 0);
}

sum(1, 2, 3);       // 6
sum(1, 2, 3, 4, 5); // 15
```

### 与普通参数结合

```ts
function greet(greeting: string, ...names: string[]): string {
  return `${greeting}, ${names.join(" and ")}!`;
}

greet("Hello", "Tom");           // "Hello, Tom!"
greet("Hello", "Tom", "Jerry");  // "Hello, Tom and Jerry!"
```

### 剩余参数必须是最后一个

```ts
// ✅ 正确
function fn(a: string, ...rest: string[]) {}

// ❌ 错误：剩余参数必须是最后一个参数
// function fn(...rest: string[], b: string) {}
```

### 元组中的剩余参数

```ts
type StringNumberBooleans = [string, number, ...boolean[]];

const a: StringNumberBooleans = ["hello", 1];
const b: StringNumberBooleans = ["hello", 1, true];
const c: StringNumberBooleans = ["hello", 1, true, false, true];
```

## 类型安全

### 解构时类型推断

```ts
interface User {
  name: string;
  age: number;
}

const user: User = { name: "Tom", age: 30 };

// TypeScript 自动推断类型
const { name, age } = user;
// name: string, age: number
```

### 重命名时的类型

```ts
const { name: userName, age: userAge }: User = user;
// userName: string
// userAge: number
```

### 解构未知类型

```ts
function process(data: unknown) {
  if (typeof data === "object" && data !== null) {
    // 使用类型守卫 + 解构
    const { name, age } = data as { name: string; age: number };
    console.log(name, age);
  }
}
```

### 函数参数解构类型

```ts
// 接口方式
interface Config {
  host: string;
  port: number;
  ssl?: boolean;
}

function connect({ host, port, ssl = false }: Config) {
  // ...
}

// 内联类型
function connect({ host, port, ssl = false }: {
  host: string;
  port: number;
  ssl?: boolean;
}) {
  // ...
}
```

## 实际应用示例

### 函数选项模式

```ts
interface RequestOptions {
  method?: "GET" | "POST" | "PUT" | "DELETE";
  headers?: Record<string, string>;
  body?: unknown;
  timeout?: number;
}

function request(url: string, options: RequestOptions = {}) {
  const {
    method = "GET",
    headers = {},
    body,
    timeout = 5000
  } = options;

  // 实现请求逻辑
  console.log(`${method} ${url}`, { headers, body, timeout });
}

request("/api/users");
request("/api/users", { method: "POST", body: { name: "Tom" } });
```

### 配置合并

```ts
const defaultConfig = {
  server: {
    host: "localhost",
    port: 3000,
  },
  database: {
    host: "localhost",
    port: 5432,
    name: "app",
  },
  logging: {
    level: "info" as const,
  },

function createConfig(userConfig: Partial<typeof defaultConfig>) {
  return {
    server: { ...defaultConfig.server, ...userConfig.server },
    database: { ...defaultConfig.database, ...userConfig.database },
    logging: { ...defaultConfig.logging, ...userConfig.logging },
  };

```

### 状态更新（React 风格）

```ts
interface State {
  user: {
    name: string;
    age: number;
  };
  loading: boolean;
  error: string | null;
}

function reducer(state: State, action: { type: string; payload?: any }): State {
  switch (action.type) {
    case "SET_NAME":
      return {
        ...state,
        user: { ...state.user, name: action.payload }
      };
    case "SET_LOADING":
      return {
        ...state,
        loading: action.payload
      };
    case "SET_ERROR":
      return {
        ...state,
        error: action.payload,
        loading: false
      };
    default:
      return state;
  }
}
```

### 数组操作

```ts
const tasks = [
  { id: 1, title: "Task 1", done: false },
  { id: 2, title: "Task 2", done: true },
  { id: 3, title: "Task 3", done: false },
];

// 添加到开头
const addedFirst = [{ id: 0, title: "New", done: false }, ...tasks];

// 添加到末尾
const addedLast = [...tasks, { id: 4, title: "New", done: false }];

// 删除元素
const removed = tasks.filter(t => t.id !== 2);

// 更新元素
const updated = tasks.map(t =>
  t.id === 2 ? { ...t, done: false } : t
);

// 移动元素
function moveItem<T>(arr: T[], from: number, to: number): T[] {
  const [item] = arr.splice(from, 1);
  const result = [...arr];
  result.splice(to, 0, item);
  return result;
}
```

## 与 Java/C# 的对比

| 特性 | Java | C# | TypeScript |
|------|------|-----|------------|
| 解构赋值 | 不支持（需手动） | 支持元组解构 | 全面支持 |
| var (int a, int b) = ... | - | ✅ | - |
| var (a, b) = tuple | - | ✅ | ✅ |
| 对象解构 | 不支持 | 不支持 | ✅ |
| 展开运算符 | 不支持 | 不支持 | ✅ |
| 剩余参数 | 不支持（需数组） | params | ✅ ...rest |
| 模式匹配 | switch 增强 | switch 表达式 | 解构 + switch |

**C# 解构示例**：
```csharp
// C# 元组解构
var (name, age) = ("Tom", 30);

// C# 带位置的解构
var person = new Person("Tom", 30);
var (pName, pAge) = person;
```

**TypeScript 对比**：
```ts
// TypeScript 更灵活
const user = { name: "Tom", age: 30, email: "tom@example.com" };
const { name, age } = user; // 直接解构对象属性
const { name: n, ...rest } = user; // 重命名 + 剩余属性
```

## 最佳实践

### 1. 解构时提供默认值

```ts
// ✅ 好的做法
function fetchUser({ id, includeInactive = false, limit = 10 }: {
  id: string;
  includeInactive?: boolean;
  limit?: number;
}) {
  // ...
}

// ❌ 避免依赖调用者提供所有参数
```

### 2. 展开运算符用于不可变更新

```ts
// ✅ 不可变更新
const updatedUsers = users.map(user =>
  user.id === id ? { ...user, name: newName } : user
);

// ❌ 直接修改
users.find(u => u.id === id).name = newName;
```

### 3. 避免过度解构

```ts
// ❌ 过度解构，可读性差
const { data: { results: { items: { 0: firstItem } } } } = response;

// ✅ 分步解构
const { data } = response;
const { results } = data;
const { items } = results;
const firstItem = items[0];

// 或按需解构
const { data } = response;
const firstItem = data.results.items[0];
```

### 4. 注意浅拷贝问题

```ts
// ❌ 浅拷贝可能导致问题
const newState = { ...state };
newState.nested.value = newValue; // 修改了原对象的嵌套属性

// ✅ 深层更新
const newState = {
  ...state,
  nested: { ...state.nested, value: newValue }
};
```

### 5. 使用剩余属性排除敏感信息

```ts
interface User {
  id: number;
  name: string;
  email: string;
  password: string;
}

function toPublicUser(user: User) {
  const { password, ...publicUser } = user;
  return publicUser;
}
```

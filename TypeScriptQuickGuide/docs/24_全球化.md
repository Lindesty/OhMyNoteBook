# 全球化处理

JavaScript 内置了 `Intl` 对象，提供国际化 API，功能类似 C# 的 `System.Globalization`。现代浏览器和 Node.js 都原生支持这些 API。

## 字符串编码

JavaScript/TypeScript 内部使用 **UTF-16** 编码存储字符串。这会导致一些常见问题：

### length 属性的陷阱

```ts
const emoji = "😀";
console.log(emoji.length); // 2（不是 1！）

const chinese = "𠮷"; // 一个汉字
console.log(chinese.length); // 2（不是 1！）

// 正确获取字符数
function getCharCount(str: string): number {
  return [...str].length;
}

console.log(getCharCount("😀")); // 1
console.log(getCharCount("𠮷")); // 1
```

### 字符串遍历

```ts
const str = "a😀b";

// ❌ 错误方式（会拆分代理对）
for (let i = 0; i < str.length; i++) {
  console.log(str[i]); // 'a', '\uD83D', '\uDE00', 'b'
}

// ✅ 正确方式
for (const char of str) {
  console.log(char); // 'a', '😀', 'b'
}

// ✅ 使用扩展运算符
const chars = [...str]; // ['a', '😀', 'b']
```

### 码点操作

```ts
const emoji = "😀";

// 获取码点
console.log(emoji.codePointAt(0)); // 128512

// 从码点创建字符
console.log(String.fromCodePoint(128512)); // "😀"

// 安全地获取字符
function charAt(str: string, index: number): string {
  return [...str][index] ?? "";
}
```

## Intl.DateTimeFormat - 日期格式化

### 基本用法

```ts
const date = new Date();

// 不同地区的日期格式
console.log(new Intl.DateTimeFormat("zh-CN").format(date)); // 2024/3/19
console.log(new Intl.DateTimeFormat("en-US").format(date)); // 3/19/2024
console.log(new Intl.DateTimeFormat("de-DE").format(date)); // 19.3.2024
console.log(new Intl.DateTimeFormat("ja-JP").format(date)); // 2024/3/19
console.log(new Intl.DateTimeFormat("ar-SA").format(date)); // ١٠/٩/١٤٤٥
```

### 详细格式化选项

```ts
const date = new Date();

const formatter = new Intl.DateTimeFormat("zh-CN", {
  year: "numeric",      // "numeric" | "2-digit"
  month: "long",        // "numeric" | "2-digit" | "long" | "short" | "narrow"
  day: "numeric",       // "numeric" | "2-digit"
  weekday: "long",      // "long" | "short" | "narrow"
  hour: "2-digit",      // "numeric" | "2-digit"
  minute: "2-digit",    // "numeric" | "2-digit"
  second: "2-digit",    // "numeric" | "2-digit"
  timeZone: "Asia/Shanghai",
  hour12: false         // 24小时制
});

console.log(formatter.format(date));
// 2024年3月19日星期二 14:30:00
```

### 时区处理

```ts
const date = new Date();

// 不同时区
const tzFormatter = (tz: string) => new Intl.DateTimeFormat("en-US", {
  hour: "numeric",
  minute: "numeric",
  timeZone: tz
});

console.log(tzFormatter("Asia/Shanghai").format(date));   // 2:30 PM
console.log(tzFormatter("America/New_York").format(date)); // 1:30 AM
console.log(tzFormatter("Europe/London").format(date));    // 6:30 AM
```

### formatToParts - 分解格式化结果

```ts
const date = new Date();
const parts = new Intl.DateTimeFormat("zh-CN", {
  year: "numeric",
  month: "long",
  day: "numeric"
}).formatToParts(date);

// [
//   { type: "year", value: "2024" },
//   { type: "literal", value: "年" },
//   { type: "month", value: "3" },
//   { type: "literal", value: "月" },
//   { type: "day", value: "19" },
//   { type: "literal", value: "日" }
// ]

// 自定义格式化
function customFormat(date: Date, locale: string): string {
  const parts = new Intl.DateTimeFormat(locale, {
    year: "numeric",
    month: "2-digit",
    day: "2-digit"
  }).formatToParts(date);

  const get = (type: string) => parts.find(p => p.type === type)?.value ?? "";

  return `${get("year")}-${get("month")}-${get("day")}`;
}
```

## Intl.Collator - 字符串比较

### 排序

```ts
const names = ["张三", "李四", "王五", "赵六"];

// 中文拼音排序
console.log(names.sort(new Intl.Collator("zh-CN").compare));
// ['李四', '王五', '张三', '赵六']

// 英文排序
const words = ["banana", "Apple", "cherry", "apple"];
console.log(words.sort(new Intl.Collator("en").compare));
// ['Apple', 'apple', 'banana', 'cherry']
```

### 忽略大小写和音调

```ts
const words = ["café", "Cafe", "CAFÉ", "cafe"];

// sensitivity 选项
// "base": 忽略大小写和音调
// "accent": 忽略大小写，区分音调
// "case": 区分大小写，忽略音调
// "variant": 区分所有差异（默认）

console.log(words.sort(new Intl.Collator("en", {
  sensitivity: "base"
}).compare));
// ['café', 'cafe', 'Cafe', 'CAFÉ']（视为相同）

// 检查相等
const collator = new Intl.Collator("en", { sensitivity: "base" });
console.log(collator.compare("café", "cafe") === 0); // true
```

### 搜索

```ts
function searchInArray<T extends string>(arr: T[], query: string): T[] {
  const collator = new Intl.Collator("en", {
    sensitivity: "base",
    usage: "search"
  });

  return arr.filter(item =>
    collator.compare(item.slice(0, query.length), query) === 0
  );
}

const fruits = ["Apple", "apple", "APPLE", "Banana", "Apricot"];
console.log(searchInArray(fruits, "app")); // ['Apple', 'apple', 'APPLE']
```

## Intl.PluralRules - 复数处理

### 基本用法

```ts
const pr = new Intl.PluralRules("en");

console.log(pr.select(0));   // "other"
console.log(pr.select(1));   // "one"
console.log(pr.select(2));   // "other"
console.log(pr.select(5));   // "other"

// 中文没有复数变化
const prZh = new Intl.PluralRules("zh-CN");
console.log(prZh.select(1)); // "other"
console.log(prZh.select(2)); // "other"
```

### 不同语言的复数规则

```ts
// 俄语有更多复数形式
const prRu = new Intl.PluralRules("ru");
console.log(prRu.select(1));   // "one"   - 1 файл
console.log(prRu.select(2));   // "few"   - 2 файла
console.log(prRu.select(5));   // "many"  - 5 файлов
console.log(prRu.select(21));  // "one"   - 21 файл

// 阿拉伯语
const prAr = new Intl.PluralRules("ar");
console.log(prAr.select(0));   // "zero"
console.log(prAr.select(1));   // "one"
console.log(prAr.select(2));   // "two"
console.log(prAr.select(3));   // "few"
console.log(prAr.select(11));  // "many"
```

### 实际应用

```ts
interface PluralForms {
  zero?: string;
  one: string;
  two?: string;
  few?: string;
  many?: string;
  other: string;
}

function pluralize(count: number, forms: PluralForms, locale: string): string {
  const rule = new Intl.PluralRules(locale).select(count);
  const form = forms[rule] ?? forms.other;
  return `${count} ${form}`;
}

// 英文
console.log(pluralize(1, { one: "item", other: "items" }, "en"));    // "1 item"
console.log(pluralize(5, { one: "item", other: "items" }, "en"));    // "5 items"

// 俄文
console.log(pluralize(1, {
  one: "файл", few: "файла", many: "файлов", other: "файлов"
}, "ru")); // "1 файл"

console.log(pluralize(2, {
  one: "файл", few: "файла", many: "файлов", other: "файлов"
}, "ru")); // "2 файла"

console.log(pluralize(5, {
  one: "файл", few: "файла", many: "файлов", other: "файлов"
}, "ru")); // "5 файлов"
```

## Intl.RelativeTimeFormat - 相对时间

### 基本用法

```ts
const rtf = new Intl.RelativeTimeFormat("zh-CN", {
  numeric: "auto"  // "always" | "auto"
});

console.log(rtf.format(-1, "day"));   // 昨天
console.log(rtf.format(0, "day"));    // 今天
console.log(rtf.format(1, "day"));    // 明天
console.log(rtf.format(-2, "week"));  // 2周前
console.log(rtf.format(3, "month"));  // 3个月后
console.log(rtf.format(-1, "year"));  // 去年
```

### 不同语言

```ts
const rtfEn = new Intl.RelativeTimeFormat("en", { numeric: "auto" });
console.log(rtfEn.format(-1, "day"));   // yesterday
console.log(rtfEn.format(1, "day"));    // tomorrow
console.log(rtfEn.format(-2, "week"));  // 2 weeks ago

const rtfJa = new Intl.RelativeTimeFormat("ja", { numeric: "auto" });
console.log(rtfJa.format(-1, "day"));   // 昨日
console.log(rtfJa.format(1, "day"));    // 明日

const rtfDe = new Intl.RelativeTimeFormat("de", { numeric: "auto" });
console.log(rtfDe.format(-1, "day"));   // gestern
console.log(rtfDe.format(1, "day"));    // morgen
```

### 实用函数

```ts
function formatRelativeTime(date: Date, locale: string): string {
  const now = new Date();
  const diffMs = date.getTime() - now.getTime();
  const diffSec = Math.round(diffMs / 1000);
  const diffMin = Math.round(diffSec / 60);
  const diffHour = Math.round(diffMin / 60);
  const diffDay = Math.round(diffHour / 24);
  const diffWeek = Math.round(diffDay / 7);
  const diffMonth = Math.round(diffDay / 30);
  const diffYear = Math.round(diffDay / 365);

  const rtf = new Intl.RelativeTimeFormat(locale, { numeric: "auto" });

  if (Math.abs(diffSec) < 60) return rtf.format(diffSec, "second");
  if (Math.abs(diffMin) < 60) return rtf.format(diffMin, "minute");
  if (Math.abs(diffHour) < 24) return rtf.format(diffHour, "hour");
  if (Math.abs(diffDay) < 7) return rtf.format(diffDay, "day");
  if (Math.abs(diffWeek) < 4) return rtf.format(diffWeek, "week");
  if (Math.abs(diffMonth) < 12) return rtf.format(diffMonth, "month");
  return rtf.format(diffYear, "year");
}

console.log(formatRelativeTime(new Date(Date.now() + 5000), "zh-CN"));   // 5秒后
console.log(formatRelativeTime(new Date(Date.now() - 3600000), "zh-CN")); // 1小时前
console.log(formatRelativeTime(new Date(Date.now() + 86400000), "zh-CN")); // 明天
```

## Intl.ListFormat - 列表格式化


## Intl.Segmenter - 文本分段

### 按词分段

```ts
const text = "Hello, 世界！这是一段测试文本。";
const segmenter = new Intl.Segmenter("zh-CN", { granularity: "word" });

const words = [...segmenter.segment(text)]
  .filter(s => s.isWordLike)
  .map(s => s.segment);

console.log(words);
// ["Hello", "世界", "这", "是", "一", "段", "测试", "文本"]
```

### 按句子分段

```ts
const text = "Hello world. This is a test. 这也是一个测试。";
const segmenter = new Intl.Segmenter("zh-CN", { granularity: "sentence" });

const sentences = [...segmenter.segment(text)].map(s => s.segment);

console.log(sentences);
// ["Hello world. ", "This is a test. ", "这也是一个测试。"]
```

### 按字符分段（正确处理代理对）

```ts
const text = "a😀b𠮷c";
const segmenter = new Intl.Segmenter("zh-CN", { granularity: "grapheme" });

const chars = [...segmenter.segment(text)].map(s => s.segment);

console.log(chars);
// ["a", "😀", "b", "𠮷", "c"]

// 对比传统方式
console.log(text.length); // 7
console.log(chars.length); // 5
```

## 最佳实践

### 1. 检测用户语言

```ts
function getUserLocale(): string {
  // 浏览器环境
  if (typeof navigator !== "undefined") {
    return navigator.language || "en-US";
  }
  // Node.js 环境
  if (process.env.LANG) {
    return process.env.LANG.split(".")[0].replace("_", "-");
  }
  return "en-US";
}
```

### 2. 缓存格式化器

```ts
// ❌ 每次创建新实例
function formatDate(date: Date) {
  return new Intl.DateTimeFormat("zh-CN").format(date);
}

// ✅ 复用实例
const dateFormatter = new Intl.DateTimeFormat("zh-CN");
function formatDate(date: Date) {
  return dateFormatter.format(date);
}

// ✅ 使用缓存
const formatters = new Map<string, Intl.DateTimeFormat>();
function getCachedFormatter(locale: string) {
  if (!formatters.has(locale)) {
    formatters.set(locale, new Intl.DateTimeFormat(locale));
  }
  return formatters.get(locale)!;
}
```

### 3. 正确处理 Unicode

```ts
// ❌ 错误方式
function reverseString(str: string): string {
  return str.split("").reverse().join("");
}
reverseString("a😀b"); // "b😀a"（代理对被拆分）

// ✅ 正确方式
function reverseString(str: string): string {
  return [...str].reverse().join("");
}
reverseString("a😀b"); // "b😀a"

// ✅ 使用 Intl.Segmenter
function reverseString(str: string): string {
  const segmenter = new Intl.Segmenter(undefined, { granularity: "grapheme" });
  return [...segmenter.segment(str)]
    .map(s => s.segment)
    .reverse()
    .join("");
}
```

### 4. 服务端 vs 客户端一致性

```ts
// 注意：服务端和客户端可能使用不同的 locale
// 建议显式指定 locale 而非依赖系统默认

const LOCALE = "zh-CN";

export function formatDate(date: Date): string {
  return new Intl.DateTimeFormat(LOCALE).format(date);
}
```
